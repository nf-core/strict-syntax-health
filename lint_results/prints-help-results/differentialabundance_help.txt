[base] Input/output options
  --study_name                              [string]  A string identifier used to name result files in the output directory [default: study] 
  --study_type                              [string]  Input data format category used for input validation and routing (not for selecting analysis 
methods).  (accepted: rnaseq, generic_matrix, affy_array, maxquant, geo_soft_file) [default: 
rnaseq]  
  --input                                   [string]  Path to CSV/TSV file containing information about the samples in the experiment. 
  --contrasts                               [string]  A CSV/TSV/YML/YAML file describing sample contrasts to compare groups. 
  --outdir                                  [string]  The output directory where the results will be saved. You have to use absolute paths to storage on 
Cloud infrastructure.  
  --round_digits                            [integer] To how many digits should numeric output in different modules be rounded? If -1 or null, will not 
round. [default: -1]  
  --seed                                    [integer] Global seed for stochastic methods 

[base] Abundance values
  --matrix                                  [string] TSV/CSV-format abundance matrix 
  --feature_length_matrix                   [string] (RNA-seq only): optional transcript/gene length matrix with samples and transcript_ids/gene_ids as 
in the abundance matrix.  
  --affy_cel_files_archive                  [string] Alternative to matrix: a compressed CEL files archive such as often found in GEO 
  --querygse                                [string] Use SOFT files from GEO by providing the GSE study identifier 

[base] Observations (e.g. samples) options
  --observations_id_col                     [string] Column in the sample sheet to be used as the primary sample identifier [default: sample] 
  --observations_type                       [string] Type of observation [default: sample] 
  --observations_name_col                   [string] Column in the sample sheet to be used as identifier for observations. If unset, the 
--observations_id_col is used.  

[base] Features options
  --features_id_col                         [string]  Feature ID attribute in the abundance table as well as in the GTF file (e.g. the gene_id field) 
[default: gene_id]  
  --features_name_col                       [string]  Feature name attribute in the abundance table as well as in the GTF file (e.g. the gene symbol 
field) [default: gene_name]  
  --features_type                           [string]  Type of feature. Often 'gene' [default: gene] 
  --sizefactors_from_controls               [boolean] When set, use the control features in scaling/ normalization (currently only supported for 
differential_method deseq2) [default: false]  
  --control_features                        [string]  A text file listing technical features (e.g. spikes) 
  --features_metadata_cols                  [string]  Comma-separated string, specifies feature metadata columns to be used for exploratory analysis, 
platform-specific [default: gene_id,gene_name,gene_biotype]  
  --features                                [string]  Supply your own feature annotations. Can be derived from the GTF (rnaseq) or from the Bioconductor 
annotation package (affy arrays).  

[base] Paramsheet options
  --paramset_name                           [string] Name of the paramset to run. In profile mode, set by the analysis profile for output directory 
naming. In paramsheet mode, selects which paramset(s) to run (comma-separated).  
  --paramsheet                              [string] Path to a paramsheet YAML file. Setting this activates multi-run (paramsheet) mode where paramsheet 
values take priority over CLI flags.

  --affy_rm_mask                            [boolean] should the spots marked as 'MASKS' set to NA? [default: false] 
  --affy_rm_outliers                        [boolean] should the spots marked as 'OUTLIERS' set to NA? [default: false]

  --filtering_min_samples_not_na            [number]           Minimum observations that must have a value (not NA) to retain the row/ feature (e.g. gene).
